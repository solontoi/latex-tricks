% Syntax for using the wallpaper package to make a specified
% .pdf file the background of the page. I use this for
% writing letters of recommendation on a .pdf of the
% college letterhead for example.

\documentclass[11pt]{letter}
\usepackage{wallpaper}

\ULCornerWallPaper{1}{./letterhead.pdf}

\begin{document}
Brilliant words go here.
\end{document}
